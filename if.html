if文について

    ソースコード（bodyの中のみ抜粋）
    
        <h1>if文について</h1>
        <p><input type="text" id="textbox"></p>
        <p><button type="button" id="start-btn">ボタン</button></p>
        <p id="target"></p>
        <script>
            'use strict';
    
            // idがtargetの要素を取得
            const target = document.getElementById('target');
    
            // ボタンがクリックされた時の処理
            document.getElementById('start-btn').addEventListener('click', e => {
                // テキストボックスに入力された値を取得して数値に変換
                let value = Number(document.getElementById('textbox').value);
                let msg = '';

                // 条件によって処理を分けたい場合にはifを使う
                if(value === 10 ) {
                    msg = '入力された値は10です。';
                } else if(value >= 10) {
                    msg = '入力された値は10以上です。';
                } else if(value <= 10) {
                    msg = '入力された値は10以下です。';
                } else {
                    msg = '入力された値は数値ではありません。';
                }
                target.textContent = msg;
            });
        </script>
