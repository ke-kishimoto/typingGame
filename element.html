要素の取得と画面への表示

        ソースコード
        
            <h1>要素の取得と画面への表示</h1>
            <p id="target1"></p>
            <p id="target2"></p>
            <script>
                'use strict';  // ←文法チェックをしてくれる。決まり文句だと思ってください
    
                // スラッシュ２つはコメント。処理に影響のないもの
    
                // 要素の取得
                // idがtarget1要素を取得して、p1という変数に入れる
                const p1 = document.getElementById('target1');
                // idがtarget2要素を取得して、p2という変数に入れる
                const p2 = document.getElementById('target2');
    
                // 画面への表示
                // target1の要素のコンテンツにHelloという文字を表示する
                p1.textContent = 'Hello'; 
                // target2の要素のコンテンツに100という数値を表示する
                p2.textContent = 100;
            
            </script>
