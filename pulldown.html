プルダウンからの値の取得

色 

    ソースコード
    
        <h1>プルダウンからの値の取得</h1>
        <p>色
            <select id="color">
                <option value="red">赤</option>
                <option value="blue">青</option>
                <option value="green">緑</option>
            </select>
        </p>
        <p><button type="button" id="select-btn">選択</button></p>
        <p id="target"></p>
        <script>
            'use strict';

            const target = document.getElementById('target');

            // ゲームスタート時の処理
            let selectBtn = () => {

                // プルダウンの値取得
                let color = document.getElementById('color').value;

                target.textContent = color;
            };

            // ボタン押下時にゲームスタート処理を実行
            document.getElementById('select-btn').addEventListener('click', selectBtn);
        </script>
