変数について

            ソースコード
            
                <h1>変数について</h1>
                <p id="target1"></p>
                <p id="target2"></p>
                <p id="target3"></p>
                <p id="target3"></p>
                <script>
                    'use strict';
        
                    // 要素を取得
                    const target1 = document.getElementById('target1');
                    const target2 = document.getElementById('target2');
                    const target3 = document.getElementById('target3');
        
                    // target1のコンテンツにnumの値をセットする
                    // numという変数に10という値を代入する
                    let num = 10;
                    target1.textContent = num;
                    
                    // numに10をプラスする
                    num = num + 10;
                    target2.textContent = num;
        
                    // 文字を扱う場合は''で囲む
                    let msg = 'Hello';
                    target3.textContent = msg;
        
                </script>
