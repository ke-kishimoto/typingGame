        ソースコード
        
            <!DOCTYPE html>
            <html lang="ja">
            <head>
                <meta charset="UTF-8">
                <title>HTMLの基本1</title>
                <style>
                    /* CSSはここに書きます */
                </style>
            </head>
            <body>
                <!-- Webページに表示したい内容はbodyタグの中に書きます -->
                <!-- これはコメント。Webページ上には表示されない -->
            
                <!-- 見出しを表します -->
                <h1>文字の表示</h1>
            
                <!-- 段落を表します -->
                <p>おはよう</p>
            
                <!-- JavaScriptで要素を取得するためにはid属性をつけます -->
                <p id="Hello">こんにちは</p>
            
                <script>
                    // JavaScriptはこの中に書きます
                </script>
            </body>
            </html>
