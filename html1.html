HTMLの基本を理解しよう

HTMLの基本を理解するためのページです

    ソースコード
    
        <!DOCTYPE html>
        <html lang="ja">
        <head>
            <meta charset="UTF-8">
            <title>HTMLの基本</title>
            <style>
                /* CSSはここに書きます */
            </style>
        </head>
        <body>
            <!-- Webページに表示したい内容はbodyタグの中に書きます -->
            <!-- これはコメント。Webページ上には表示されない -->
        
            <h1>HTMLの基本を理解しよう</h1>
        
            <!-- pで囲った文章や要素は改行される -->
            <p>HTMLの基本を理解するためのページです</p>
        
            <!-- ボタンを表示する -->
            <p><button type="button">ボタン</button></p>
        
            <script>
                // JavaScriptはこの中に書きます
            </script>
        </body>
        </html>
