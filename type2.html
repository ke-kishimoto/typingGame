キーボードで入力したキーが繋がって表示されます

            ソースコード（bodyの中のみ抜粋）
            
                <h1>キーボードで入力したキーが繋がって表示されます</h1>
                <p id="target"></p>
                <script>
                    'use strict';
        
                    // idがtargetの要素を取得
                    const target = document.getElementById('target');
        
                    // キー押下時の処理
                    document.addEventListener('keydown', e => {
                        // 入力した文字をつなげる
                        target.textContent = target.textContent + e.key;  
                    });
                </script>
