ランダムな数値を取得します

        ソースコード（bodyの中のみ抜粋）
        
            <h1>ランダムな数値を取得します</h1>
            <p><button type="button" id="start-btn" name="start-btn">スタート</button></p>
            <p id="target1"></p>
            <p id="target2"></p>
            <p id="target3"></p>
            <script>
                'use strict';
    
                const target1 = document.getElementById('target1');
                const target2 = document.getElementById('target2');
                const target3 = document.getElementById('target3');
    
                // ボタンがクリックされた時の処理
                document.getElementById('start-btn').addEventListener('click', e => {
    
                    target1.textContent = Math.random();  // 0~1のランダムな値(小数) 
                    target2.textContent = Math.random() * 10;  // 0~10のランダムな値(小数)
                    target3.textContent = Math.floor(Math.random() * 10);  // 0~10のランダムな値(整数)
                });
            </script>
