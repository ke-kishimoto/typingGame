タイマー

    
        <h1>タイマー</h1>
        <p>
            <button type="button" id="start-btn">スタート</button>
            <button type="button" id="stop-btn">ストップ</button>
        </p>
        <p id="time"></p>
        <script>
            'use strict';

            const time = document.getElementById('time');
            let setTimeoutId = 0;
            let startTime = 0;

            function dispTime() {
                time.textContent = ((new Date - startTime) / 1000).toFixed(2) + '秒';
                setTimeoutId = setTimeout(() => {
                    dispTime();
                }, 50);
            }

            // スタートボタン押下時の処理
            document.getElementById('start-btn').addEventListener('click', e => {
                startTime = new Date();
                dispTime();
            });

            // ストップボタン押下時の処理
            document.getElementById('stop-btn').addEventListener('click', e => {
                clearTimeout(setTimeoutId);
            });

        </script>
