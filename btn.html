ボタン押下時の処理

    ソースコード（bodyの中のみ抜粋）
    
        <h1>ボタン押下時の処理</h1>
        <p><button type="button" id="start-btn">スタート</button></p>
        <p id="target"></p>
        <script>
            'use strict';

            // idがtargetの要素を取得
            const target = document.getElementById('target');

            // ボタンがクリックされた時の処理
            document.getElementById('start-btn').addEventListener('click', e => {
                target.textContent = 'ゲームスタート';
            });
        </script>
